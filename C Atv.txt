#include <stdio.h>

int main()
{
    float a, b, c, areaA, areaB, areaC;
    printf("Digite o valor de a: ");
    scanf("%f", &a);
    printf("Digite o valor de b: ");
    scanf("%f", &b);
    printf("Digite o valor de c: ");
    scanf("%f", &c);
    printf("Valor de a: %.4f\n", a);
    printf("Valor de b: %.4f\n", b);
    printf("Valor de c: %.4f\n", c);
    areaA = a*a;
    areaB = ((a*a) + (b*b))*(1/2.0);
    areaC = ((a + b) * c) / 2;
    printf("Area A (Quadrado): %.4f\n", areaA);
    printf("Area B (Triang. Retangulo): %.4f\n", areaB);
    printf("Area C (Trapezio): %.4f\n", areaC);
    return 0;
}

#include <stdio.h>

int main()
{
    int segundos;
    double horas, minutos, segundos_restantes;
    printf("Digite o valor em segundos: ");
    scanf("%d", &segundos);
    horas = segundos / 3600.0;
    minutos = (segundos % 3600) / 60.0;
    segundos_restantes = segundos % 60;
    printf("Horas: %.0f, Minutos: %.0f, Segundos: %.0f\n", horas, minutos, segundos_restantes);
    return 0;
}
