8. #include <stdio.h>

int main() { 
    float celcius, fahrenheit;
    printf("Deseja converter de Celsius para Fahrenheit ou de Fahrenheit para Celsius? (C/F): ");
    char choice;
    scanf(" %c", &choice);
    if (choice == 'C' || choice == 'c') {
        printf("Digite a temperatura em Celsius: ");
        scanf("%f", &celcius);
        fahrenheit = (celcius * 9/5) + 32;
        printf("%.2f Celsius é igual a %.2f Fahrenheit\n", celcius, fahrenheit);
    } else if (choice == 'F' || choice == 'f') {
        printf("Digite a temperatura em Fahrenheit: ");
        scanf("%f", &fahrenheit);
        celcius = (fahrenheit - 32) * 5/9;
        printf("%.2f Fahrenheit é igual a %.2f Celsius\n", fahrenheit, celcius);
    } else {
        printf("Opção inválida. Por favor, escolha 'C' ou 'F'.\n");
    }
    return 0;
}
